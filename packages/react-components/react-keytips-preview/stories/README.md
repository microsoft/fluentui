## Usage

\`\`\`js
module.exports = {
stories: ['../packages/react-components/react-keytips-preview/stories/src/**/*.stories.mdx', '../packages/react-components/react-keytips-preview/stories/src/**/index.stories.@(ts|tsx)'],
}
\`\`\`
